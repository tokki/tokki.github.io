Certainly! Let's create a new model for the billing table to record user balance changes. We'll call this model "Transaction" to represent each balance change event. Here's the data model for the Transaction table:

<GO>
package models

import (
	"gorm.io/gorm"
	"time"
)

type TransactionType string

const (
	Deposit  TransactionType = "deposit"
	Withdraw TransactionType = "withdraw"
	Charge   TransactionType = "charge"
	Refund   TransactionType = "refund"
)

type Transaction struct {
	gorm.Model
	UserID      uint            `gorm:"not null" json:"user_id"`
	User        User            `gorm:"foreignKey:UserID" json:"-"`
	Amount      float64         `gorm:"not null" json:"amount"`
	Type        TransactionType `gorm:"not null" json:"type"`
	Description string          `json:"description"`
	Balance     float64         `gorm:"not null" json:"balance"` // Balance after this transaction
}

func (t *Transaction) BeforeCreate(tx *gorm.DB) (err error) {
	t.CreatedAt = time.Now()
	return
}

Let's break down this model:

  1. UserID: References the User who owns this transaction.
  2. User: A GORM association to the User model.
  3. Amount: The amount of the transaction (positive for deposits/refunds, negative for withdrawals/charges).
  4. Type: An enum-like field to categorize the transaction (deposit, withdraw, charge, or refund).
  5. Description: A text field for additional information about the transaction.
  6. Balance: The user's balance after this transaction is applied.

We've also defined a TransactionType type with constants for different types of transactions.

Now, let's update the User model to include a relationship with transactions:

<GO>
type User struct {
	gorm.Model
	Username     string        `gorm:"uniqueIndex;not null" json:"username"`
	Email        string        `gorm:"uniqueIndex;not null" json:"email"`
	Password     string        `gorm:"not null" json:"-"`
	FullName     string        `json:"full_name"`
	IsAdmin      bool          `gorm:"default:false" json:"is_admin"`
	Balance      float64       `gorm:"default:0" json:"balance"`
	Transactions []Transaction `gorm:"foreignKey:UserID" json:"-"`
}

We've added a Transactions field to the User model to establish a one-to-many relationship between Users and Transactions.

  2. Categorize transactions by type.

  4. Keep track of the user's balance at each transaction point.

You can now create functions to handle transactions, update user balances, and query transaction history. For example:

<GO>
func CreateTransaction(userID uint, amount float64, transactionType TransactionType, description string) (*Transaction, error) {
	var user User
	if err := DB.First(&user, userID).Error; err != nil {
		return nil, err
	}

	newBalance := user.Balance + amount
	transaction := Transaction{
		UserID:      userID,
		Amount:      amount,
		Type:        transactionType,
		Description: description,
		Balance:     newBalance,
	}

	err := DB.Transaction(func(tx *gorm.DB) error {
		if err := tx.Create(&transaction).Error; err != nil {
			return err
		}
		if err := tx.Model(&user).Update("Balance", newBalance).Error; err != nil {
			return err
		}
		return nil
	})

	if err != nil {
		return nil, err
	}

	return &transaction, nil
}

<GO>
package handlers

import (
	"net/http"
	"your_project/models"
	"golang.org/x/crypto/bcrypt"
	"gorm.io/gorm"
	"github.com/gin-gonic/gin"
)

var db *gorm.DB

func SetDB(database *gorm.DB) {
	db = database
}

func RenderTemplate(c *gin.Context, templateName string, data gin.H) {
	if data == nil {
		data = gin.H{}
	}
	data["user"] = c.MustGet("user")
	c.HTML(http.StatusOK, templateName, data)
}

func Index(c *gin.Context) {
	RenderTemplate(c, "index.html", gin.H{"title": "Home"})
}

func Login(c *gin.Context) {
	if c.Request.Method == "GET" {
		RenderTemplate(c, "login.html", gin.H{"title": "Login"})
		return
	}

	username := c.PostForm("username")
	password := c.PostForm("password")

	var user models.User
	if err := db.Where("username = ?", username).First(&user).Error; err != nil {
		c.HTML(http.StatusBadRequest, "login.html", gin.H{"error": "Invalid credentials"})
		return
	}

	if err := bcrypt.CompareHashAndPassword([]byte(user.Password), []byte(password)); err != nil {
		c.HTML(http.StatusBadRequest, "login.html", gin.H{"error": "Invalid credentials"})
		return
	}

	// Set session or JWT token here
	c.Redirect(http.StatusSeeOther, "/")
}

func Register(c *gin.Context) {
	if c.Request.Method == "GET" {
		RenderTemplate(c, "register.html", gin.H{"title": "Register"})
		return
	}

	username := c.PostForm("username")
	email := c.PostForm("email")
	password := c.PostForm("password")
	fullname := c.PostForm("fullname")

	hashedPassword, _ := bcrypt.GenerateFromPassword([]byte(password), bcrypt.DefaultCost)

	user := models.User{
		Username: username,
		Email:    email,
		Password: string(hashedPassword),
		FullName: fullname,
	}

	if err := db.Create(&user).Error; err != nil {
		c.HTML(http.StatusBadRequest, "register.html", gin.H{"error": "Unable to register"})
		return
	}

	c.Redirect(http.StatusSeeOther, "/login")
}

func Admin(c *gin.Context) {
	var users []models.User
	db.Find(&users)
	RenderTemplate(c, "admin.html", gin.H{"title": "Admin Panel", "users": users})
}

func Logout(c *gin.Context) {
	// Clear session or JWT token here
	c.Redirect(http.StatusSeeOther, "/")
}
